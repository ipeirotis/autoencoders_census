```typescript
app.post('/api/upload', requireAuth, uploadLimiter, upload.single('file'), async (req, res, next) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: 'No file uploaded' });
    }

    // Validate CSV content
    const validation = await validateCSVContent(req.file.buffer);
    if (!validation.valid) {
      return res.status(400).json({ error: 'File validation failed' });
      // Don't expose validation.reason to client (logged server-side)
    }

    // Generate safe filename (discard user-provided name)
    const safeFilename = generateSafeFilename(req.user.id);

    // ... rest of upload logic with safeFilename
  } catch (error) {
    next(error); // Pass to error handler
  }
});
```

```typescript
describe('Security Integration Tests', () => {
  describe('Authentication', () => {
    it('rejects unauthenticated upload-url request', async () => {
      const res = await request(app).post('/api/jobs/upload-url');
      expect(res.status).toBe(401);
    });

    it('accepts authenticated upload-url request', async () => {
      // Login first, use session cookie
      const res = await authenticatedAgent.post('/api/jobs/upload-url')
        .send({ filename: 'test.csv' });
      expect(res.status).toBe(200);
    });
  });

  describe('Rate Limiting', () => {
    it('blocks 6th upload in 15 minutes', async () => {
      // Make 5 successful requests
      for (let i = 0; i < 5; i++) {
        await authenticatedAgent.post('/api/jobs/upload-url').send({ filename: 'test.csv' });
      }
      // 6th should be blocked
      const res = await authenticatedAgent.post('/api/jobs/upload-url').send({ filename: 'test.csv' });
      expect(res.status).toBe(429);
    });
  });

  describe('Input Validation', () => {
    it('rejects non-UUID job ID', async () => {
      const res = await authenticatedAgent.get('/api/jobs/job-status/not-a-uuid');
      expect(res.status).toBe(400);
      expect(res.body.error).toBe('Validation failed');
    });
  });

  describe('CORS', () => {
    it('blocks unauthorized origin', async () => {
      const res = await request(app)
        .get('/api/ping')
        .set('Origin', 'https://evil.com');
      expect(res.headers['access-control-allow-origin']).toBeUndefined();
    });
  });

  describe('Security Headers', () => {
    it('includes X-Frame-Options header', async () => {
      const res = await request(app).get('/api/ping');
      expect(res.headers['x-frame-options']).toBe('DENY');
    });
  });

  describe('Error Handling', () => {
    it('hides stack trace in production', async () => {
      process.env.NODE_ENV = 'production';
      // Trigger an error
      const res = await authenticatedAgent.get('/api/jobs/job-status/nonexistent-but-valid-uuid');
      expect(res.body.stack).toBeUndefined();
      process.env.NODE_ENV = 'test';
    });
  });
});
```
